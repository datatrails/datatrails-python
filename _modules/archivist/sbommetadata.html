    def from_dict(cls, indict):
        """Ignore unexpected fields"""
        params = inspect_signature(cls).parameters
        diff = set(indict) - set(params)
        if diff:
            LOGGER.info("WARN: extra keys %s ignored", diff)
        return cls(**{k: indict[k] for k in params})
