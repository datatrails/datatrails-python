def create_compliance_policy(arch):
    """
    Creates a PERIOD OUTSTANDING compliance policy for feeding one's dog. If RKVST sees a "Feed"
    event without a closing "Fed" event within an hour, then Golden Retriever Asset is
    out of compliance.
    """
    period_outstanding_policy = arch.compliance_policies.create(
        CompliancePolicyPeriodOutstanding(
            description="Ensuring my dog is fed within an hour",
            display_name="Feeding My Dog Hourly",
            asset_filter=[
                ["attributes.arc_display_type=Golden Retriever"],
            ],
            event_display_type="Feed",
            closing_event_display_type="Fed",
            time_period_seconds=3600,
        )
    )
    print("PERIOD_OUTSTANDING_POLICY:", json_dumps(period_outstanding_policy, indent=4))
    return period_outstanding_policy
