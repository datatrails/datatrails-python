def create_access(arch):
    """
    Pre-requisite: Ensure that RKVST Subject IDS have been exchanged with parties that one
    wants to share with BEFORE creating this Access Policy.

    How to exchange RKVST Subject IDs:
    https://docs.rkvst.com/docs/rkvst-basics/sharing-assets-with-obac/

    Creates an Access Policy that shares read only data for Artists with other RKVST tenancies
    """
    props = {
        "display_name": "Sharing Artist Asset with Record Labels",
        "description": "Sharing Artist Asset with Record Labels",
    }
    filters = [{"or": ["attributes.arc_display_type=Artists"]}]
    access_permissions = [
        {
            "asset_attributes_read": ["*"],
            "asset_attributes_write": [],
            "behaviours": ASSET_BEHAVIOURS,
            "event_arc_display_type_read": ["*"],
            "event_arc_display_type_write": [],
            "include_attributes": [],
            "subjects": ["subjects/34b291c3-30f4-4d89-9ec7-85f57354f798"],
            "user_attributes": [],
        }
    ]

    return arch.access_policies.create(props, filters, access_permissions)
